REVERSAL OF STRING


DECLARE STR1 AS STRING 
DECLARE STR2 AS STRING 
DECLARE CHAR AS CHAR 
DECLARE COUNT AS INTEGER 

STR1 = ""
STR2 = ""
CHAR = ""
COUNT = 0     

  FOR COUNT = 1 TO LEN(STR1)
   CHAR = MID(STR1,COUNT,1)
   STR2 = CHAR & STR2 
  NEXT

OUTPUT " REVERSED STRING: " & STR2
