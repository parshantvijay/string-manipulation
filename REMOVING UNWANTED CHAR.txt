DECLARE STR1 AS STRING 
DECLARE CHAR AS CHAR 
DECLARE NEXTCHAR AS CHAR 
DECLARE COUNT AS INTEGER 
DECLARE STR2 AS STRING 

STR1 = ""
STR2 = ""
CHAR = ''
NEXTCHAR = ''
COUNT = 0

OUTPUT "ENTER CORRUPTED TXT: "
INPUT STR1

OUTPUT "ENTER CHARACTER TO REMOVE: "
INPUT CHAR

FOR COUNT = 1 TO LEN(STR1)
  NEXTCHAR = MID(STR1,COUNT,1)
    IF NEXTCHAR<>CHAR THEN 
    STR2 = STR2 & NEXTCHAR 
    END IF 
NEXT 

OUTPUT "TXT AFTER CORRECTION: " & STR2
